真正的处理错误

func writeFile2(filename string) {
    file, err := os.OpenFile(
        filename, os.O_EXCL|os.O_CREATE, 0666)
    if err != nil {
        if pathError, ok := err.(*os.PathError); !ok {
            // 真的 不知道是森什么错了
            // 那就挂掉程序吧
            panic(err)
        } else {
            fmt.Printf("%s, %s, %s\n",
                pathError.Op,
                pathError.Path,
                pathError.Err,
            )
        }
        return
    }
    defer file.Close()
    writer := bufio.NewWriter(file)
    defer writer.Flush()

    for i := 0; i < 20; i++ {
        fmt.Fprintln(writer, i)
    }
}
